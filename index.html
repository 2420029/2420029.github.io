Option Explicit

Sub 請求先キー生成バッチ()

    Dim wsIn As Worksheet, wsOPA As Worksheet, wsWBS As Worksheet, wsOut As Worksheet
    Dim lastRow As Long, i As Long, outRow As Long
    Dim dict As Object
    Dim key As String
    Dim orderVal As String, itemVal As String
    
    Set wsIn = Worksheets("INPUT")
    Set wsOPA = Worksheets("OPA検索")
    Set wsWBS = Worksheets("WBS検索")
    Set wsOut = Worksheets("MASTER")
    
    Set dict = CreateObject("Scripting.Dictionary")
    
    '========================
    ' ① 入力取得 + 重複排除
    '========================
    lastRow = wsIn.Cells(wsIn.Rows.Count, 1).End(xlUp).Row
    
    For i = 2 To lastRow
        orderVal = wsIn.Cells(i, 1).Value
        itemVal = wsIn.Cells(i, 2).Value
        
        key = orderVal & "|" & itemVal
        
        If Not dict.exists(key) Then
            dict.Add key, Array(orderVal, itemVal)
        End If
    Next i
    
    '========================
    ' ② シートクリア
    '========================
    wsOPA.Range("A13:F10000").ClearContents
    wsWBS.Range("A13:F10000").ClearContents
    wsOut.Cells.Clear
    
    '========================
    ' ③ OPA投入
    '========================
    i = 13
    Dim k As Variant
    
    For Each k In dict.keys
        wsOPA.Cells(i, 1).Value = dict(k)(0)
        wsOPA.Cells(i, 2).Value = dict(k)(1)
        i = i + 1
    Next k
    
    '========================
    ' ④ OPAマクロ実行
    '========================
    Application.Run "opamain"
    
    '========================
    ' ⑤ WBS投入
    '========================
    i = 13
    For Each k In dict.keys
        wsWBS.Cells(i, 1).Value = dict(k)(0)
        wsWBS.Cells(i, 2).Value = dict(k)(1)
        i = i + 1
    Next k
    
    '========================
    ' ⑥ WBSマクロ実行
    '========================
    Application.Run "wbsmain"
    
    '========================
    ' ⑦ 出力ヘッダ
    '========================
    wsOut.Cells(1, 1) = "order"
    wsOut.Cells(1, 2) = "item"
    wsOut.Cells(1, 3) = "system"
    wsOut.Cells(1, 4) = "billing_key"
    
    outRow = 2
    i = 13
    
    '========================
    ' ⑧ 判定
    '========================
    For Each k In dict.keys
        
        orderVal = dict(k)(0)
        itemVal = dict(k)(1)
        
        Dim opaRes As String, wbsRes As String
        
        opaRes = wsOPA.Cells(i, 6).Value
        wbsRes = wsWBS.Cells(i, 6).Value
        
        If opaRes = "利用可" And wbsRes <> "利用可" Then
            wsOut.Cells(outRow, 1) = orderVal
            wsOut.Cells(outRow, 2) = itemVal
            wsOut.Cells(outRow, 3) = "OPA"
            wsOut.Cells(outRow, 4) = orderVal & "-" & itemVal
        
        ElseIf wbsRes = "利用可" And opaRes <> "利用可" Then
            wsOut.Cells(outRow, 1) = orderVal
            wsOut.Cells(outRow, 2) = itemVal
            wsOut.Cells(outRow, 3) = "WBS"
            wsOut.Cells(outRow, 4) = orderVal & "-" & itemVal
        
        Else
            ' エラー
            wsOut.Cells(outRow, 1) = orderVal
            wsOut.Cells(outRow, 2) = itemVal
            wsOut.Cells(outRow, 3) = "ERROR"
            wsOut.Cells(outRow, 4) = ""
        End If
        
        outRow = outRow + 1
        i = i + 1
        
    Next k
    
    MsgBox "完了しました"

End Sub
